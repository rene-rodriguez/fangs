# v1 UI Polish Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use `superpowers:subagent-driven-development` (recommended) or `superpowers:executing-plans` to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Make Fangs feel like a finished v1 app by polishing toasts, notification-ring feedback, and the settings modal, plus wiring agent rings to native desktop notifications when the app is unfocused.

**Architecture:** Keep all changes inside the existing immediate-mode Raylib render loop. Toast and ring-pulse state lives in small structs in the UI modules and is driven by `frame_dt_sec` from `main.c` (never `GetFrameTime()`). The settings modal keeps its single-file implementation but renders as a vertical-tabbed form. The existing `draft -> cfg` save path and `desktop_notify_agent_ring()` seam are reused unchanged.

**Tech Stack:** C11, Raylib/Raygui, existing `g_ui_theme` color tokens, `frame_dt_sec` from `main.c`.

- [ ] **Step 1: Implement `toast_draw()` in `src/ui_toast.c`**

Add at the bottom of `src/ui_toast.c`, above `toast_clear`:

```c
static Color toast_accent(ToastLevel level)
{
    switch (level) {
        case TOAST_INFO:  return UI2RAY(g_ui_theme.accent);
        case TOAST_WARN:  return UI2RAY(g_ui_theme.warn);  // fall back to orange-ish
        case TOAST_ERROR: return UI2RAY(g_ui_theme.danger); // or a hard-coded red if missing
    }
    return UI2RAY(g_ui_theme.accent);
}

static Color toast_fill(void)
{
    Color c = UI2RAY(g_ui_theme.panel_bg);
    c.a = 240;
    return c;
}

static void draw_toast_icon(float cx, float cy, float size, ToastLevel level, Color fg)
{
    float r = size * 0.5f;
    if (level == TOAST_INFO) {
        DrawCircleV((Vector2){cx, cy}, r, fg);
        DrawTextEx(GetFontDefault(), "i", (Vector2){cx - r * 0.22f, cy - r * 0.55f}, size * 0.9f, 0.0f, WHITE);
    } else if (level == TOAST_WARN) {
        Vector2 v1 = {cx, cy - r};
        Vector2 v2 = {cx - r * 0.9f, cy + r * 0.8f};
        Vector2 v3 = {cx + r * 0.9f, cy + r * 0.8f};
        DrawTriangle(v1, v2, v3, fg);
        DrawTextEx(GetFontDefault(), "!", (Vector2){cx - r * 0.18f, cy - r * 0.25f}, size * 0.8f, 0.0f, WHITE);
    } else {
        DrawCircleV((Vector2){cx, cy}, r, fg);
        DrawTextEx(GetFontDefault(), "x", (Vector2){cx - r * 0.28f, cy - r * 0.5f}, size * 0.85f, 0.0f, WHITE);
    }
}

void toast_draw(Font font, float scale, double dt)
{
    (void)dt; // age is already advanced by toast_tick; kept for API symmetry

    int n = toast_count();
    if (n == 0) return;

    float s = (scale > 0.1f) ? scale : 1.0f;
    int screen_w = GetScreenWidth();
    int screen_h = GetScreenHeight();

    const float toast_w = 360.0f * s;
    const float margin = 12.0f * s;
    const float gap = 8.0f * s;
    const float row_pad = 12.0f * s;
    const float icon_w = 26.0f * s;
    const float font_size = 15.0f * s;
    const float enter_dur = 0.15;
    const float exit_dur = 0.25;
    const float corner = 6.0f * s;
    const float shadow_off = 3.0f * s;

    float y = margin; // top-right, stack downward

    for (int i = 0; i < n; i++) {
        ToastLevel level;
        const char *msg;
        float alpha;
        if (!toast_get(i, &level, &msg, &alpha)) continue;

        // Resolve the entry so we can read its birth.
        int idx = (g_toast.head + g_toast.count - 1 - i) % TOAST_RING;
        double age = g_toast.ring[idx].birth;
        double ttl = g_toast.ring[idx].ttl;
        double max_ttl = g_toast.ring[idx].max_ttl;

        Vector2 msz = MeasureTextEx(font, msg, font_size, 0.0f);
        float text_h = msz.y;
        float toast_h = (text_h < icon_w ? icon_w : text_h) + row_pad * 2.0f;

        // Enter/exit animation.
        float anim = 1.0f;
        if (age < enter_dur) {
            float t = (float)(age / enter_dur);
            anim = 0.95f + 0.05f * (t * t); // scale
            alpha *= t * t;                  // fade in
        } else if (ttl < exit_dur) {
            float t = (float)(ttl / exit_dur);
            anim = 1.0f + (1.0f - t) * 0.05f; // slight scale up while leaving
            alpha *= t;                       // linear fade out
        }

        float draw_w = toast_w * anim;
        float draw_x = (float)screen_w - margin - draw_w;

        Rectangle rect = {draw_x, y, draw_w, toast_h};

        // Shadow.
        DrawRectangleRounded((Rectangle){rect.x + shadow_off, rect.y + shadow_off, rect.width, rect.height}, corner / rect.height, 8, (Color){0, 0, 0, (uint8_t)(48 * alpha)});

        // Fill.
        Color fill = toast_fill();
        fill.a = (uint8_t)(fill.a * alpha);
        DrawRectangleRounded(rect, corner / rect.height, 8, fill);

        // Left accent.
        Color sev = toast_accent(level);
        sev.a = (uint8_t)(sev.a * alpha);
        DrawRectangleRounded((Rectangle){rect.x, rect.y, 4.0f * s, rect.height}, 2.0f * s / rect.height, 4, sev);

        // Icon.
        float ix = rect.x + row_pad + icon_w * 0.5f;
        float iy = rect.y + rect.height * 0.5f;
        draw_toast_icon(ix, iy, icon_w, level, sev);

        // Text.
        Color fg = UI2RAY(g_ui_theme.text);
        fg.a = (uint8_t)(fg.a * alpha);
        float text_x = rect.x + row_pad + icon_w + 8.0f * s;
        float text_y = rect.y + (rect.height - text_h) * 0.5f;
        DrawTextEx(font, msg, (Vector2){text_x, text_y}, font_size, 0.0f, fg);

        // Progress bar along bottom edge.
        if (max_ttl > 0.0 && ttl > 0.0) {
            float p = (float)(ttl / max_ttl);
            float bar_h = 2.0f * s;
            float bar_w = rect.width * p;
            Color bar = sev;
            bar.a = (uint8_t)(bar.a * 0.7f);
            DrawRectangleRounded((Rectangle){rect.x, rect.y + rect.height - bar_h, bar_w, bar_h}, bar_h / 2.0f, 4, bar);
        }

        y += toast_h + gap;
    }
}
```

If `g_ui_theme.warn`/`g_ui_theme.danger` do not exist, hard-code warm colors for warn/error or add the tokens to `ui_theme.h`/`ui_theme.c` as part of this task.
